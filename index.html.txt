        // 1. 获取 URL 里的 id 参数
        const params = new URLSearchParams(window.location.search);
        const id = params.get('id');

        // 2. ⚠️ 关键点：这里必须和 Xcode 里的 URL Schemes 一模一样！
        // 如果你 Xcode 里改成了 'kitchensurvivor'，这里就用 'kitchensurvivor'
        // 如果你 Xcode 还是 'kitchen'，这里就用 'kitchen'
        const scheme = 'kitchen'; 
        
        const deepLink = `${scheme}://recipe?id=${id || ''}`;
        
        // 设置按钮链接
        document.getElementById('open-btn').href = deepLink;

        // 3. 检测是否在微信里
        const isWeChat = /MicroMessenger/i.test(navigator.userAgent);
        
        if (isWeChat) {
            // 如果是微信，显示遮罩，提示用户去 Safari 打开
            document.getElementById('wechat-tip').style.display = 'block';
        } else {
            // 如果不是微信（比如 Safari），尝试自动跳转
            window.location.href = deepLink;
        }
    </script>
